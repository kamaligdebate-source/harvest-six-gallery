async function loadGallery() { try { const res = await fetch(sheetUrl); const csv = await res.text(); const lines = csv.split('\n').slice(1); allData = lines.map(line => { // This Regex specifically handles commas inside double quotes correctly const c = line.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); const clean = (val) => val ? val.replace(/^"|"$/g, '').trim() : ""; if (!c[0]) return null; return { name: clean(c[0]), batchNum: clean(c[1]), accolade: clean(c[2]), batchName: clean(c[3]), cohort: clean(c[4]), institution: clean(c[5]), breaks: parseInt(clean(c[6])) || 0 }; }).filter(i => i !== null); render(); } catch (e) { console.error("Error loading field data:", e); } }
